📝 CAPL 基本结构 入门

/* CAPL 程序结构 */ variables { int g_counter = 0; float g_speed = 0.0; message 0x100 EngineMsg; // 声明报文 } on start { write("程序启动"); } /* 报文事件 */ on message 0x200 { float speed = this.byte(1); write("收到: %d", speed); }

📤 报文发送与接收 入门

/* 周期发送 - 定时器方式 */ variables { msTimer CyclicTx; message 0x200 DriveMsg; } on start { setTimer(CyclicTx, 100); // 100ms周期 } on timer CyclicTx { DriveMsg.DLC = 8; DriveMsg.byte(0) = 0x01; DriveMsg.byte(1) = 0x64; // 100 km/h output(DriveMsg); // 发送 setTimer(CyclicTx, 100); }

🎮 环境变量与Panel交互 入门

/* 环境变量 - 与CANoe Panel联动 */ variables { float envThrottle; // 油门 0-100% float envBrake; // 刹车 } on envVar envThrottle { float throttle = getValue(envThrottle); write("油门: %.1f%%", throttle); } /* 按键响应 */ on key 'p' { putValue(envGear, 0); } on key 'd' { putValue(envGear, 3); }

🔍 UDS诊断功能 诊断

/* 诊断请求 */ on key 'r' { diagRequest UDS.ReadDTCByStatus_req req; req.send(); } on diagResponse UDS.ReadDTCByStatus_req { if (this.Result == 0) { write("诊断成功!"); } else { write("失败: %d", this.Result); } }

⚙️ 状态机实现 进阶

variables { enum DriveState { IDLE, DRIVE, BRAKE }; DriveState g_state = IDLE; } on message 0x200 { byte cmd = this.byte(0); switch(g_state) { case IDLE: if (cmd == 0x01) g_state = DRIVE; break; case DRIVE: if (cmd == 0x02) g_state = BRAKE; break; case BRAKE: if (this.byte(1) == 0) g_state = IDLE; break; } }
