📤 报文发送与接收 入门

// 周期发送报文
variables
{
    msTimer CyclicTx;
    message 0x200 DriveMsg;
}

on start
{
    setTimer(CyclicTx, 100); // 100ms周期
}

on timer CyclicTx
{
    DriveMsg.DLC = 8;
    DriveMsg.byte(0) = 0x01;
    DriveMsg.byte(1) = 0x64; // 100 km/h
    output(DriveMsg);
    setTimer(CyclicTx, 100);
}

// 接收报文
on message 0x300
{
    float speed = this.Speed;  // 访问dbc信号
    write("收到速度: %.1f", speed);
}

🎮 环境变量交互 入门

// 环境变量 - 与Panel交互
variables
{
    float envThrottle;  // 油门 0-100%
    float envBrake;     // 刹车 0-100%
    byte   envGear;      // 档位 0-3
}

on envVar envThrottle
{
    float throttle = getValue(envThrottle);
    
    // 发送到CAN总线
    message 0x200 msg;
    msg.ThrottlePos = throttle;
    output(msg);
    
    write("油门: %.1f%%", throttle);
}

// 按键控制
on key 'p' { putValue(envGear, 0); }
on key 'd' { putValue(envGear, 3); }

🔍 UDS 诊断功能 诊断

variables
{
    diagRequest Door_Control_req;
}

// 发送诊断请求
on key 'r'
{
    // 读DTC (故障码)
    diagRequest UDS.ReadDTCByStatus_req req;
    req.GetMultiByteValue(0xDTCStatusMask, 0xFF);
    req.send();
}

// 诊断响应处理
on diagResponse UDS.ReadDTCByStatus_req
{
    if (this.Result == 0)
    {
        write("诊断成功!");
        byte numDTC = this.GetMultiByteValue(0xDTCNumber);
        write("DTC数量: %d", numDTC);
    }
    else
    {
        write("诊断失败: %d", this.Result);
    }
}

⚙️ 状态机实现 进阶

variables
{
    enum DriveState { 
        IDLE, DRIVE, BRAKE, FAULT 
    };
    DriveState g_state = IDLE;
}

on message 0x200
{
    byte cmd = this.byte(0);
    
    switch(g_state)
    {
        case IDLE:
            if (cmd == 0x01) g_state = DRIVE;
            break;
            
        case DRIVE:
            if (cmd == 0x02) g_state = BRAKE;
            break;
            
        case BRAKE:
            if (this.byte(1) == 0)
                g_state = IDLE;
            break;
    }
}

variables
{
    float g_vehicleSpeed = 0.0;
    float g_engineSpeed = 0.0;
    int   g_gear = 0;  // 0=P, 1=R, 2=N, 3=D
    float g_throttle = 0;
    float g_brake = 0;

}

on start
{
    setTimer(EngineTx, 10);
}

on timer EngineTx
{
    message 0x100 EngineMsg;
    EngineMsg.EngineSpeed = g_engineSpeed;
    EngineMsg.ThrottlePos = g_throttle;
    output(EngineMsg);
    setTimer(EngineTx, 10);
}

on envVar envThrottle
{
    g_throttle = getValue(envThrottle);
    g_engineSpeed = g_vehicleSpeed * 50 + g_throttle * 100;
}
