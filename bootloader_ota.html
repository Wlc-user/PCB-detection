0x34 请求下载

// 请求
0x34
0x00           // dataFormatIdentifier (0x00=无压缩无加密)
                    // bit7-4: 压缩方法, bit3-0: 加密方法
0x44           // addressAndLengthFormatIdentifier
                    // bit7-4: 地址字节数 (4=32位), bit3-0: 尺寸字节数
00 00 10 00  // memoryAddress (起始地址 0x00100000)
00 00 04 00  // memorySize (大小 256KB = 0x40000)

// 响应
0x74
20 00        // numberOfBlocks (块数量 = 0x2000)
00 10        // blockLength (每块4字节granularity×0x40 = 0x1000)

0x36 数据传输

// 请求 (每个数据块)
0x36
0x01           // blockSequenceCounter (1~numberOfBlocks)
XX XX XX...  // transferRequestParameterRecord (数据)

// 响应
0x76
0x01           // blockSequenceCounter (回显)

// 示例: 传输1024字节 (分4块)
TX: 36 01 00 01 02 03 04 05 06 07...  // 256B
RX: 76 01
TX: 36 02 08 09 0A 0B 0C 0D 0E 0F...  // 256B
RX: 76 02
TX: 36 03 10 11 12 13 14 15 16 17...  // 256B
RX: 76 03
TX: 36 04 18 19 1A 1B 1C 1D 1E 1F...  // 256B
RX: 76 04

Bootloader 状态机

enum BootState {
    APP_RUNNING,       // 应用运行中
    BOOT_REQUEST,      // 收到跳转请求
    BOOT_ACTIVE,       // Bootloader激活
    PROGRAMMING,       // 编程中
    VERIFICATION,      // 验证中
    APP_VALID,         // 应用有效
    APP_INVALID        // 应用无效
};

// 状态转移
APP_RUNNING ──(0x10 02)──→ BOOT_ACTIVE
BOOT_ACTIVE ──(0x34)────→ PROGRAMMING
PROGRAMMING ──(0x37)────→ VERIFICATION
VERIFICATION ─(成功)────→ APP_VALID
VERIFICATION ─(失败)────→ BOOT_ACTIVE
APP_VALID ────(复位)────→ APP_RUNNING

安全机制

安全项说明实现
会话验证必须在编程会话0x10 0x02
安全解锁Level 2或更高0x27 Seed-Key
完整性校验CRC32/Checksum传输后验证
版本检查禁止降级SW.ver > current
指纹记录记录编程者DID 0xF195
加密传输防止窃取0x34 dataFormat

完整CAPL实现

// Bootloader OTA 测试脚本
variables
{
    int g_currentBlock = 0;
    int g_totalBlocks = 0;
    byte g_blockData[4096];
}

// 主流程
on start
{
    write("开始OTA升级...");
    
    // Step 1: 切换到编程会话
    sendSessionControl(0x02);
    testWaitForTimeout(500);
    
    // Step 2: 安全解锁
    securityAccess(0x03);
    testWaitForTimeout(500);
    
    // Step 3: 请求下载
    int blockCount = requestDownload(
        0x00100000,  // 地址
        0x40000     // 大小
    );
    
    // Step 4: 传输数据
    for (int i=1; i<=blockCount; i++)
    {
        prepareBlockData(i, g_blockData);
        transferData(i, g_blockData);
        write("传输块 %d/%d", i, blockCount);
    }
    
    // Step 5: 退出传输
    requestTransferExit();
    
    // Step 6: ECU复位
    ecuReset(0x01);
    
    write("OTA升级完成!");
}

testfunction requestDownload(dword addr, dword size)
{
    message 0x7E0 req;
    req.byte(0) = 0x34;
    req.byte(1) = 0x00;  // 无压缩加密
    req.byte(2) = 0x44;  // 4字节地址+4字节大小
    req.byte(3) = (addr >> 24) & 0xFF;
    req.byte(4) = (addr >> 16) & 0xFF;
    req.byte(5) = (addr >> 8) & 0xFF;
    req.byte(6) = addr & 0xFF;
    req.byte(7) = (size >> 24) & 0xFF;
    req.byte(8) = (size >> 16) & 0xFF;
    req.byte(9) = (size >> 8) & 0xFF;
    req.byte(10) = size & 0xFF;
    req.dlc = 11;
    output(req);
    
    // 等待响应
    message * resp;
    testWaitForMessage(resp, 0x7E8, 3000);
    
    if (resp.byte(0) == 0x74)
    {
        int blockLen = (resp.byte(3) << 8) | resp.byte(4);
        return size / blockLen;
    }
    return 0;
}
