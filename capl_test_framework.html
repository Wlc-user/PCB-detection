📝 testcase 基本用法 基础

// ==================== testcase 定义 ====================

// 最简单的测试用例
testcase TC_Basic()
{
    testStep("Step 1", "发送唤醒报文");
    message 0x01 wakeup;
    output(wakeup);
    
    // 等待响应，超时1000ms
    message * resp;
    int result = testWaitForMessage(resp, 0x100, 1000);
    
    // 断言检查
    if (result == 1)
    {
        testStepPass("收到ECU响应");
    }
    else
    {
        testStepFail("ECU无响应");
    }
}

// ==================== 带参数的测试用例 ====================

testcase TC_WithParameters(int speed, int timeout)
{
    message 0x200 cmd;
    cmd.Speed = speed;
    output(cmd);
    
    testWaitForTimeout(timeout);
    testCheck(g_actualSpeed == speed, "速度验证");
}

✅ 断言与检查函数 基础

// ==================== 常用断言函数 ====================

// 通用检查
testCheck(condition, description);
// 示例
testCheck(speed > 0, "速度应为正数");

// 数值范围检查
testCheckOrWarn(rpm >= 800 && rpm <= 7000, "转速范围检查");

// 精确相等 (带容差)
testCheckEqual(actual, expected, tolerance, description);
// 示例：允许±5的误差
testCheckEqual(actualTemp, 25.0, 5.0, "温度检查");

// 信号值检查
testAddSignalCheck(signalName, min, max);
testAddSignalCheck("EngineSpeed", 0, 8000);

// ==================== 等待与超时 ====================

// 等待指定报文
message * rxMsg;
int ret = testWaitForMessage(rxMsg, 0x100, 1000);
// 返回值: 1=收到, 0=超时

// 等待条件满足
int ret = testWaitForCondition(speed > 100, 5000);

// 等待信号值
testWaitForSignal("EngineSpeed", 5000, 500, 50);

// ==================== 数据驱动测试 ====================

struct SpeedTestData
{
    byte gear;
    int targetSpeed;
    int maxTime;
};

const SpeedTestData speedTable[] = {
    {1, 30,  2000},
    {2, 60,  3000},
    {3, 100, 4000}
};

testcase TC_SpeedTableDriven()
{
    for (int i = 0; i < elcount(speedTable); i++)
    {
        testStep("档位 %d", speedTable[i].gear);
        int elapsed = measureResponseTime(
            speedTable[i].gear, speedTable[i].targetSpeed
        );
        testCheck(elapsed < speedTable[i].maxTime, "响应时间");
    }
}

📦 Test Unit 模块化测试 框架

// ==================== TestUnit.can ====================

testunit "DiagnosticUnit"

// 对外暴露的测试函数
testfunction Diag_ReadDTC(int dtcMask)
{
    diagRequest UDS.ReadDTCByStatus_req req;
    req.GetMultiByteValue(0xDTCStatusMask, dtcMask);
    req.send();
}

testfunction Diag_SwitchSession(byte sessionType)
{
    diagRequest UDS.SessionControl_req req;
    req.GetMultiByteValue(0xSF, sessionType);
    req.send();
}

// ==================== 主测试文件调用 ====================

includes
{
    #include "DiagnosticUnit.can"
}

testcase TC_DiagnosticFlow()
{
    Diag_SwitchSession(0x03);
    Diag_ReadDTC(0xFF);
}
