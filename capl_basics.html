🏗️ CAPL 程序结构 入门

/* ==================== CAPL 基本结构 ==================== */

/* 1. 变量声明区 - 全局变量 */
variables {
    int    g_counter = 0;      // 整型变量
    float  g_speed   = 0.0;   // 浮点变量
    message 0x100 EngineMsg;   // 报文变量
}

/* 2. 程序启动事件 */
on start {
    write("=== CAPL 程序启动 ===");
    setTimer(myTimer, 100);  // 启动定时器
}

/* 3. 报文接收事件 */
on message 0x200 {
    float speed = this.byte(1);
    write("收到速度: %d km/h", speed);
}

/* 4. 定时器事件 */
on timer myTimer {
    write("定时器触发");
    setTimer(myTimer, 100);  // 重新设置
}

📤 周期报文发送 入门

variables {
    msTimer CyclicTx;
    message 0x200 DriveMsg;
}

on start {
    setTimer(CyclicTx, 100);  // 100ms周期
}

on timer CyclicTx {
    DriveMsg.DLC = 8;
    DriveMsg.byte(0) = 0x01;
    DriveMsg.byte(1) = 100;     // 速度
    DriveMsg.byte(2) = 3000;    // 转速
    output(DriveMsg);           // 发送!
    setTimer(CyclicTx, 100);
}

📥 报文接收处理 入门

/* 接收所有报文 */
on message * {
    int id = this.CAN;
    int dlc = this.DLC;
    write("ID:0x%x DLC:%d", id, dlc);
}

/* 接收特定报文 */
on message 0x300 {
    byte cmd = this.byte(0);
    word data = this.word(2);
    
    if (cmd == 0x01) {
        write("指令: 启动");
    }
}

🎮 环境变量与Panel交互 入门

variables {
    float envThrottle;  // 油门 0-100%
    float envBrake;     // 刹车
    int    envGear;      // 档位
}

/* 环境变量变化事件 */
on envVar envThrottle {
    float throttle = getValue(envThrottle);
    write("油门: %.1f%%", throttle);
}

/* 按键响应 */
on key 'p' { putValue(envGear, 0); }  // P档
on key 'd' { putValue(envGear, 3); }  // D档
on key 'r' { putValue(envGear, 1); }  // R档

⏱️ 定时器使用 入门

msTimer (毫秒)

variables {
    msTimer myTimer;
}

on start {
    setTimer(myTimer, 500);  // 500ms
}

on timer myTimer {
    // 处理逻辑
    setTimer(myTimer, 500);
}

secondTimer (秒)

variables {
    secondTimer tickTimer;
}

on start {
    setTimer(tickTimer, 5);  // 5秒
}

on timer tickTimer {
    g_counter++;
    write("计数器: %d", g_counter);
    setTimer(tickTimer, 5);
}

⚙️ 状态机实现 进阶

variables {
    enum DriveState { IDLE, DRIVE, BRAKE, FAULT };
    DriveState g_state = IDLE;
    int g_errorCount = 0;
}

/* 根据报文切换状态 */
on message 0x200 {
    byte cmd = this.byte(0);
    
    switch(g_state) {
        case IDLE:
            if (cmd == 0x01) {
                g_state = DRIVE;
                write("状态: IDLE → DRIVE");
            }
            break;
        case DRIVE:
            if (cmd == 0x02) {
                g_state = BRAKE;
                write("状态: DRIVE → BRAKE");
            }
            break;
        case BRAKE:
            if (this.byte(1) == 0) {
                g_state = IDLE;
                write("状态: BRAKE → IDLE");
            }
            break;
    }
}

/* 诊断请求发送 */
on key 'r' {
    diagRequest UDS.ReadDTCByStatus_req req;
    req.GetMultiByteValue(0xDTCStatusMask, 0xFF);
    req.send();
    write("发送诊断请求: 读取DTC");
}

/* 诊断响应处理 */
on diagResponse UDS.ReadDTCByStatus_req {
    if (this.Result == 0) {
        write("✓ 诊断成功!");
        // 解析响应数据
    } else {
        write("✗ 诊断失败: %d", this.Result);
    }
}
