🔧 UDS诊断 & 状态机

Unified Diagnostic Services · 车载诊断协议

📡 UDS 协议概述

项目说明
标准ISO 14229-1 (UDSonCAN), ISO 15765-2
寻址物理寻址 (1:1) / 功能寻址 (1:N)
会话默认会话 / 扩展会话 / 编程会话
安全安全访问 (Seed-Key)

📦 报文结构

// UDS诊断请求
Byte 0: SID (Service ID)     // 0x10 = 诊断会话控制
Byte 1+: SF (Sub-Function) // 子功能参数
Byte N+: Data           // 数据参数

// 正响应
Byte 0: SID + 0x40       // 0x50 = 会话控制响应

// 负响应
Byte 0: 0x7F
Byte 1: Rej-SID
Byte 2: NRC               // 否定响应码

🔢 常见 NRC

NRC名称含义
0x11serviceNotSupported服务不支持
0x22conditionsNotCorrect条件不满足
0x33securityAccessDenied安全访问被拒绝
0x35invalidKey密钥无效
0x78requestCorrectlyReceived等待中

📝 会话控制 0x10

// 子功能
0x01: defaultSession      // 默认会话
0x02: programmingSession  // 编程会话
0x03: extendedDiagnostic // 扩展诊断会话

// 示例
0x10 03           // 请求: 切换到扩展会话
0x50 03 00 32 00 C3 // 响应: P2=50ms, P2*=20s

🔐 安全访问 0x27

// Step 1: 请求种子
0x27 01           // 请求Level1种子
0x67 01 XX XX XX XX // ECU返回4字节随机种子

// Step 2: 发送密钥
0x27 02 YY YY YY YY // Tester计算并发送密钥
0x67 02           // 成功
0x7F 27 0x35   // 失败: 密钥无效

📝 状态转移规则

// 转移条件
Default → Extended: 0x10 0x03
Default → Programming: 0x10 0x02
Extended → Default: 0x10 0x01 或 S3超时(5s)
Extended → Programming: 0x10 0x02
Programming → Extended: 0x10 0x03

// 安全转移
Locked → Waiting: 0x27 0x01 (请求种子)
Waiting → Unlocked: 0x27 0x02 (密钥正确)
Unlocked → Locked: 会话切换 或 超时

📝 会话状态机CAPL实现

variables
{
    enum UdsSessionState {
        STATE_DEFAULT,
        STATE_EXTENDED,
        STATE_PROGRAMMING
    };
    UdsSessionState g_currentSession = STATE_DEFAULT;
    int g_securityUnlocked = 0;
    msTimer S3Timer;
}

// 会话控制处理
on message 0x7E0
{
    if (this.byte(0) == 0x10)
    {
        byte sessionType = this.byte(1);
        switch (sessionType & 0x7F)
        {
            case 0x01: g_currentSession = STATE_DEFAULT; break;
            case 0x02: g_currentSession = STATE_PROGRAMMING; break;
            case 0x03: g_currentSession = STATE_EXTENDED; break;
            default: sendNRC(0x10, 0x12);
        }
        cancelTimer(S3Timer);
        if (g_currentSession != STATE_DEFAULT)
            setTimer(S3Timer, 5000);
    }
}

// S3超时处理
on timer S3Timer
{
    g_currentSession = STATE_DEFAULT;
    g_securityUnlocked = 0;
}

// 0x3E 保持会话
on message 0x7E0
{
    if (this.byte(0) == 0x3E)
    {
        cancelTimer(S3Timer);
        setTimer(S3Timer, 5000);
    }
}

🛡️ 安全访问CAPL实现

variables
{
    enum SecurityState { SEC_LOCKED, SEC_WAITING_KEY, SEC_UNLOCKED };
    SecurityState g_securityState = SEC_LOCKED;
    byte g_seed[4];
}

on message 0x7E0
{
    if (this.byte(0) == 0x27)
    {
        byte subFunc = this.byte(1);
        byte level = subFunc & 0x7F;
        
        if (level % 2 == 1) // 请求种子
        {
            generateSeed(g_seed);
            sendSeedResponse(level, g_seed);
            g_securityState = SEC_WAITING_KEY;
        }
        else // 发送密钥
        {
            byte key[4];
            for(int i=0; i<4; i++) key[i] = this.byte(2+i);
            
            byte expected[4];
            calculateKey(g_seed, expected);
            
            if (compareKeys(key, expected))
            {
                g_securityState = SEC_UNLOCKED;
                sendPositiveResponse(0x67, subFunc);
            }
            else
            {
                sendNRC(0x27, 0x35);
            }
        }
    }
}

void calculateKey(byte seed[4], byte key[4])
{
    key[0] = (seed[0] ^ 0xA5);
    key[1] = (seed[1] ^ 0x5A);
    key[2] = (seed[2] ^ 0xC3);
    key[3] = (seed[3] ^ 0x3C);
}
